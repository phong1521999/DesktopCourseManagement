    private void btAddActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btAddActionPerformed
        // TODO add your handling code here:
        
        String code=txtAddCode.getText().trim();
        String name=txtAddName.getText().trim();
        String credit=txtAddCredit.getText().trim();
        if(isValidation(code, name, credit)){
            name=name.toLowerCase();
            name=name.substring(0,1).toUpperCase()+name.substring(1);
            parent.getCourseList().add(new Course(code, name, credit));
            JOptionPane.showMessageDialog(this, "Add successful");
            hideLabelWaring();
            txtAddCode.setText("");
            txtAddName.setText("");
            txtAddCredit.setText("");
        }
    }//GEN-LAST:event_btAddActionPerformed

    boolean isValidation(String code, String name, String credit) {
        //Code: can’t be empty, can’t be duplicated
        boolean isValidation=true;
        if(code.isEmpty()){
            lbWarningAddCode.setText("Code can't be empty");
            isValidation=false;
        }else if(findCourseByCode(code)!=null){
            lbWarningAddCode.setText("Code is existed");
            isValidation=false;
        }
        //Name can’t be empty
        if(name.isEmpty()){
            lbWarningAddName.setText("Name can't be empty");
            isValidation=false;
        }
        //credit can't be empty
        if(credit.isEmpty()){
            lbWarningAddCredit.setText("Credit can't be empty");
            isValidation=false;
        }
        //Credit must be a positive number and less than or equals to 33.
        else{
            try {
                int creditInteger=Integer.parseInt(credit);
                if(creditInteger<0){
                    lbWarningAddCredit.setText("Credit must be a positive number");
                    isValidation=false;
                }else if(creditInteger>33){
                    lbWarningAddCredit.setText("Credit must be less than or equals to 33");
                    isValidation=false;
                }
            } catch (NumberFormatException e) {
                lbWarningAddCredit.setText("Credit must be a positive number, less than or equals to 33");
                isValidation=false;
            }
        } 
            
       return isValidation;    
    }

    Course findCourseByCode(String code) {
        ArrayList<Course> courseList=parent.getCourseList();
        Course courseSearch=null;
        for (Course course : courseList) {
            if(course.getCode().equalsIgnoreCase(code)){
                courseSearch=course;
                break;
            }
        }
        return courseSearch;
    }
