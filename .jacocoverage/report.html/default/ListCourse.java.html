    public ListCourse(java.awt.Frame parent, String title) {
        super(parent, title);
        initComponents();
        this.setLocationRelativeTo(null);
        this.parent = (CourseManagement) parent;
        copyDataAndSortByCredit();

    }

    void copyDataAndSortByCredit() {
        ArrayList<Course> courseList=parent.getCourseList();
        for (Course course : courseList) {
            displayAllCoursesList.add(course);
        }
        Collections.sort(displayAllCoursesList,new Comparator<Course>(){
            @Override
            public int compare(Course o1, Course o2) {
                return o1.getCredit().compareTo(o2.getCredit());
            }
            
        });
    }
    void displayAllCourse(){
        String display="";
        for (Course course : displayAllCoursesList) {
            display+=course.getCode()+" | "+course.getName()+" | "+course.getCredit()+"\n";
        }
      //  System.out.println(display);
        txaDisplay.setText(display);
    }
